73 {
74 std::array<std::size_t, rank()> sorted_ids(natural_ids);
75 std::sort(sorted_ids.begin(), sorted_ids.end());
76 return mem_size()
77 - (0 + ...
78 + (sorted_ids[ddc::type_seq_rank_v<
79 TensorIndex,
80 ddc::detail::TypeSeq<TensorIndex...>>]
81 == TensorIndex::mem_size() - rank()
82 + ddc::type_seq_rank_v<
83 TensorIndex,
84 ddc::detail::TypeSeq<TensorIndex...>>

87 TensorIndex::mem_size()

90 ddc::detail::TypeSeq<TensorIndex...>>]
91 - 1,
92 rank()
93 - ddc::type_seq_rank_v<
94 TensorIndex,
95 ddc::detail::TypeSeq<TensorIndex...>>)))
96 - 1;

99private:
100 static constexpr bool permutation_parity(std::array<std::size_t, rank()> ids)
101 {
102 bool cnt = false;
103 for (int i = 0; i < rank(); i++)
104 for (int j = i + 1; j < rank(); j++)
105 if (ids[i] > ids[j])
106 cnt = !cnt;
107 return cnt;
108 }
109
110public:
111 KOKKOS_FUNCTION static constexpr std::size_t access_id(
112 std::array<std::size_t, rank()> const natural_ids)
113 {
114 if constexpr (rank() <= 1) {
115 return mem_id(natural_ids);
116 } else {
117 if (std::all_of(natural_ids.begin(), natural_ids.end(), [&](const std::size_t id) {
118 return id == *natural_ids.begin();
119 })) {
120 return 0;
121 } else if (!permutation_parity(natural_ids)) {
122 return 1 + mem_id(natural_ids);
123 } else {
124 return access_size() + mem_id(natural_ids);
125 }
126 }
127 }

129 KOKKOS_FUNCTION static constexpr std::size_t access_id_to_mem_id(std::size_t access_id)
130 {
131 if constexpr (rank() <= 1) {
132 return access_id;
133 } else {
134 if (access_id != 0) {
135 return (access_id - 1) % mem_size();
136 } else {
137 return std::numeric_limits<std::size_t>::max();
138 }
139 }

142 template <class Tensor, class Elem, class Id, class FunctorType>
143 KOKKOS_FUNCTION static constexpr Tensor::element_type process_access(
144 const FunctorType& access,
145 Tensor tensor,
146 Elem elem)
147 {
148 if constexpr (rank() <= 1) {
149 return access(tensor, elem);
150 } else {
151 if (elem.template uid<Id>() == 0) {
152 return 0.;
153 } else if (elem.template uid<Id>() < access_size()) {
154 return access(tensor, elem);
155 } else {
156 return -access(tensor, elem);
157 }
158 }
159 }

161 KOKKOS_FUNCTION static constexpr std::array<std::size_t, rank()>
162 mem_id_to_canonical_natural_ids(std::size_t mem_id)
163 {
164 assert(mem_id < mem_size());
165 if constexpr (rank() == 0) {
166 return std::array<std::size_t, rank()> {};
167 } else {
168 std::array<std::size_t, rank()> ids;
169 std::size_t d
170 = ddc::type_seq_element_t<0, ddc::detail::TypeSeq<TensorIndex...>>::mem_size();
171 std::size_t r = rank();
172 for (std::size_t i = 0; i < rank(); ++i) {
173 const std::size_t triangle_size = misc::binomial_coefficient(d, r - i);
174 for (std::size_t j = 0; j < d; ++j) {
175 const std::size_t subtriangle_size
176 = misc::binomial_coefficient(d - j - 1, r - i);
177 if (triangle_size - subtriangle_size > mem_id) {
178 ids[i] = ddc::type_seq_element_t<0, ddc::detail::TypeSeq<TensorIndex...>>::
179 mem_size()
180 - d + j;
181 mem_id -= triangle_size - misc::binomial_coefficient(d - j, r - i);
182 d -= j + 1;
183 break;
184 }
185 ids[i] = 0;
186 }
187 }
188 return ids;
189 }
190 }
