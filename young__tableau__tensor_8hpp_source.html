55 static constexpr std::pair<std::vector<double>, std::vector<std::size_t>> mem_lin_comb(
56 std::array<std::size_t, sizeof...(TensorIndex)> const ids)
57 {
58 std::pair<std::vector<double>, std::vector<std::size_t>> result {};
59 constexpr csr::Csr v = young_tableau::template v<
60 TensorYoungTableauIndex<YoungTableau, TensorIndex...>,
61 TensorIndex...>(ddc::DiscreteDomain<TensorIndex...>(
62 ddc::DiscreteElement<TensorIndex...>(ddc::DiscreteElement<TensorIndex>(0)...),
63 ddc::DiscreteVector<TensorIndex...>(
64 ddc::DiscreteVector<TensorIndex>(TensorIndex::size())...)));
65 for (std::size_t j = 0; j < v.values().size(); ++j) {
66 if (((v.idx()[ddc::type_seq_rank_v<TensorIndex, ddc::detail::TypeSeq<TensorIndex...>>]
67 [j]
68 == TensorIndex::access_id(ids))
69 && ...)) {
70 std::get<0>(result).push_back(v.values()[j]);
71 std::size_t k = 0;
72 while (k < v.coalesc_idx().size() - 1 && v.coalesc_idx()[k + 1] <= j) {
73 k++;
74 }
75 std::get<1>(result).push_back(k);
76 }
77 }
78 return result;
79 }
