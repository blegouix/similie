115template <class SimplexType, class LayoutStridedPolicy, class MemorySpace>
116class LocalChain
117{
118public:
119 using memory_space = MemorySpace;
120 using simplex_type = SimplexType;
121 using discrete_element_type = typename simplex_type::discrete_element_type;
122 using discrete_vector_type = typename simplex_type::discrete_vector_type;
123 static constexpr std::size_t MAX_SIZE
124 = 1UL << ddc::type_seq_size_v<ddc::to_type_seq_t<discrete_vector_type>>;
125 using storage_type = std::array<simplex_type, MAX_SIZE>;

127 using iterator_type = LocalChainIterator<SimplexType, LayoutStridedPolicy, MemorySpace>;

162 template <misc::NotSpecialization<ddc::DiscreteVector>... T>
163 KOKKOS_FUNCTION constexpr explicit LocalChain(
164 discrete_element_type origin,
165 T... simplex) noexcept
166 : m_origin(origin)
167 , m_size(sizeof...(T))
168 {
169 static_assert(sizeof...(T) <= MAX_SIZE);
170 std::size_t i = 0;
171 ((m_vects[i++] = simplex), ...);
172 assert(check() == 0 && "there are duplicate simplices in the chain");
173 if constexpr (sizeof...(T) > 1) {
174 assert(misc::are_all_equal(simplex.discrete_element()...)
175 && "LocalChain must contain simplices with same origin (if not, use Chain)");
176 }
177 assert((!simplex.negative() && ...)
178 && "negative simplices are not supported in LocalChain");

364 template <class T>
365 KOKKOS_FUNCTION constexpr LocalChain& operator*=(T t)
366 {
367 if (t == 1) {
368 } else if (t == -1) {
369 for (std::size_t i = 0; i < m_size; ++i) {
370 m_vects[i] = -m_vects[i];
371 }
372 } else {
373 assert(false && "chain must be multiplied by 1 or -1");
374 }
375 return *this;
376 }

389 return false;

392 auto other_simplex = simplices.begin();
393 for (; simplex < end(); ++simplex, ++other_simplex) {
394 if (*simplex != *other_simplex) {
395 return false;
396 }
397 }
398 return true;

404template <std::size_t K, misc::Specialization<ddc::DiscreteDomain> Dom>
405struct TangentBasis;
406
407template <std::size_t K, class... Tag>
408struct TangentBasis<K, ddc::DiscreteDomain<Tag...>>
409{
410 template <class MemorySpace = Kokkos::HostSpace, class Elem>
411 KOKKOS_FUNCTION static constexpr auto run(Elem elem)
412 {
413 using chain_type = LocalChain<Simplex<K, Tag...>, Kokkos::LayoutRight, MemorySpace>;
414 std::array<std::ptrdiff_t, sizeof...(Tag)> permutation
415 = {0 * ddc::type_seq_rank_v<Tag, ddc::detail::TypeSeq<Tag...>>...};
416 for (auto i = permutation.begin(); i < permutation.begin() + K; ++i) {
417 *i = 1;
418 }
419 chain_type basis {typename chain_type::discrete_element_type(elem)};
420 do {
421 typename chain_type::discrete_vector_type vect;
422 ddc::detail::array(vect) = permutation;
423 basis += vect;
424 } while (std::prev_permutation(permutation.begin(), permutation.end()));
425 return basis;
426 }
427};
